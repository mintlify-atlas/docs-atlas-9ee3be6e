---
title: 'Get Clan Statistics'
description: 'Retrieve comprehensive statistics for a clan and its members'
api: 'GET /clan/{groupId}'
---

Get the stats for a clan. Data updates weekly.

## Path Parameters

<ParamField path="groupId" type="string" required>
  The unique identifier for the clan. This is the Bungie.net group ID.
  
  Example: `3148408`
</ParamField>

## Response

<ResponseField name="aggregateStats" type="object" required>
  Aggregate statistics for the entire clan
  
  <Expandable title="Aggregate Stats">
    <ResponseField name="stats" type="object">
      Clan-wide statistics
      
      <Expandable title="Stats">
        <ResponseField name="clears" type="integer">
          Total number of raid completions by all clan members
        </ResponseField>
        
        <ResponseField name="averageClears" type="integer">
          Average number of clears per clan member
        </ResponseField>
        
        <ResponseField name="freshClears" type="integer">
          Total number of fresh (full) raid completions
        </ResponseField>
        
        <ResponseField name="averageFreshClears" type="integer">
          Average number of fresh clears per clan member
        </ResponseField>
        
        <ResponseField name="sherpas" type="integer">
          Total number of sherpa runs (helping first-timers)
        </ResponseField>
        
        <ResponseField name="averageSherpas" type="integer">
          Average number of sherpas per clan member
        </ResponseField>
        
        <ResponseField name="timePlayedSeconds" type="integer">
          Total time spent in raids across all clan members (in seconds)
        </ResponseField>
        
        <ResponseField name="averageTimePlayedSeconds" type="integer">
          Average time spent in raids per clan member (in seconds)
        </ResponseField>
        
        <ResponseField name="totalContestScore" type="number">
          Total contest mode score for the clan
        </ResponseField>
        
        <ResponseField name="weightedContestScore" type="number">
          Weighted contest mode score accounting for participation
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="ranks" type="object" nullable>
      Global leaderboard rankings for the clan (null if not ranked)
      
      <Expandable title="Ranks">
        <ResponseField name="clearsRank" type="integer">
          Clan's rank on the total clears leaderboard
        </ResponseField>
        
        <ResponseField name="freshClearsRank" type="integer">
          Clan's rank on the fresh clears leaderboard
        </ResponseField>
        
        <ResponseField name="sherpasRank" type="integer">
          Clan's rank on the sherpas leaderboard
        </ResponseField>
        
        <ResponseField name="timePlayedSecondsRank" type="integer">
          Clan's rank on the time played leaderboard
        </ResponseField>
        
        <ResponseField name="totalContestScoreRank" type="integer">
          Clan's rank on the total contest score leaderboard
        </ResponseField>
        
        <ResponseField name="weightedContestScoreRank" type="integer">
          Clan's rank on the weighted contest score leaderboard
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="members" type="array" required>
  Array of clan members with their individual statistics
  
  <Expandable title="Member">
    <ResponseField name="playerInfo" type="object" nullable>
      Player information (null for private profiles)
      
      <Expandable title="Player Info">
        <ResponseField name="membershipId" type="string">
          Destiny membership ID (int64 format)
        </ResponseField>
        
        <ResponseField name="membershipType" type="integer" nullable>
          Platform type (1=Xbox, 2=PSN, 3=Steam, 4=Blizzard, 5=Stadia, 6=Epic)
        </ResponseField>
        
        <ResponseField name="displayName" type="string" nullable>
          Legacy platform-specific display name
        </ResponseField>
        
        <ResponseField name="bungieGlobalDisplayName" type="string" nullable>
          Bungie name (without discriminator)
        </ResponseField>
        
        <ResponseField name="bungieGlobalDisplayNameCode" type="string" nullable>
          Bungie name discriminator code
        </ResponseField>
        
        <ResponseField name="iconPath" type="string" nullable>
          Path to player's profile icon
        </ResponseField>
        
        <ResponseField name="lastSeen" type="string">
          ISO 8601 timestamp of last activity
        </ResponseField>
        
        <ResponseField name="isPrivate" type="boolean">
          Whether the player's profile is private
        </ResponseField>
        
        <ResponseField name="cheatLevel" type="integer">
          Cheat detection level (0=None, 1=Suspicious, 2=Moderate, 3=Extreme, 4=Blacklisted)
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="stats" type="object">
      Individual member statistics
      
      <Expandable title="Stats">
        <ResponseField name="clears" type="integer">
          Total number of raid completions
        </ResponseField>
        
        <ResponseField name="freshClears" type="integer">
          Number of fresh (full) raid completions
        </ResponseField>
        
        <ResponseField name="sherpas" type="integer">
          Number of sherpa runs performed
        </ResponseField>
        
        <ResponseField name="totalTimePlayedSeconds" type="integer">
          Total time spent in raids (in seconds)
        </ResponseField>
        
        <ResponseField name="contestScore" type="number">
          Contest mode score
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Notes

- Clan statistics are updated weekly, so recent activity may not be immediately reflected
- The `playerInfo` field will be `null` for members with private profiles
- All time values are in seconds
- The `cheatLevel` indicates the integrity status of a player's account
- Rankings are only available for clans that appear on the global leaderboards
